/* Abot Kamay Brand Colors */
:root {
  --ak-sunset:   #E8772E;  /* Primary warm */
  --ak-gold:     #F5923A;  /* Warm highlight */
  --ak-amber:    #C05E1C;  /* Warm deep */
  --ak-ocean:    #2D6FE8;  /* Primary cool */
  --ak-sky:      #5B9BF5;  /* Cool highlight */
  --ak-deep-sea: #1B4FBF;  /* Cool deep */
  --ak-white:    #FAFAFA;
  --ak-black:    #06060A;
  --ak-navy:     #0C1628;
  --ak-font:     'Plus Jakarta Sans', 'Inter', sans-serif;
}

/* Gradient shortcuts */
.ak-gradient-warm {
  background: linear-gradient(135deg, #F5923A, #C05E1C);
}
.ak-gradient-cool {
  background: linear-gradient(135deg, #5B9BF5, #1B4FBF);
}
.ak-gradient-brand {
  background: linear-gradient(135deg, #E8772E, #2D6FE8);
}
6. Usage Example

    <circle cx="47" cy="40" r="26" fill="none" stroke="#2D6FE8" stroke-width="4"/>
    <circle cx="40" cy="40" r="4" fill="#FFF"/>
  </svg>

  <!-- Brand Name -->
  <span style="font-family:'Plus Jakarta Sans',sans-serif; font-weight:900;
               font-size:20px; color:#FAFAFA; letter-spacing:2px;">
    ABOT KAMAY
  </span>

</header>
